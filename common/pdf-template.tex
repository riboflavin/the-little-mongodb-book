%% Set up macro for formatting chapter numbers and chapter titles
\newcommand{\usetitlenumber}[2]
{
\clearpage
\vspace*{1in}

\fontspec[Color=bookgray]{Azuza-Bold}
\fontsize{20pt}{20pt}
\bf
\selectfont
#1. 
\fontspec[Color=bookgray]{Metric-Regular}
\fontsize{16pt}{16pt}
\bf
\selectfont
#2
\vspace*{0.25in}
\restoredefaults
\setlength{\parindent}{0in}
\\
}

%% Set up macro for formatting chapter titles alone

{
\clearpage
\vspace*{1in}

\fontspec[Color=bookgray]{Metric-Regular}
\fontsize{16pt}{16pt}
\bf
\selectfont
#1

\restoredefaults
\setlength{\parindent}{0in}
\\
}
